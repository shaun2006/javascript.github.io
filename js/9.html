    <script language="Javascript">
          function calculate()
          {
              event.preventDefault();
              var n1,n2,n3,avg,total;
              n1=parseInt(f1.nnum1.value);
              n2=parseInt(f1.nnum2.value);
              n3=parseInt(f1.nnum3.value);
              total=n1+n2+n3;  
              avg=parseFloat(total/3); 
              document.writeln("Average of entered three numbers="+avg+"<br>Sum of 3 numbers="+total);
          }
